returns varchar(150) AS
BEGIN
   return @palabra+CAST(@numero as varchar)
END

CREATE FUNCTION reponerPorX(@nombre varchar(100))
returns varchar(100) AS
BEGIN
declare @i int
declare @new varchar(100)
SET @i=1

	while @i<len(@nombre)
	BEGIN 
		SET @nombre= REPLACE(@nombre,SUBSTRING(@nombre,@i,@i),'X')
		SET @i=@i+1
	END
	return @nombre
END
GO
